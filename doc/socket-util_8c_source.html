257 /* Drain all the data currently in the receive queue of a datagram socket (and
258  * possibly additional data). There is no way to know how many packets are in
259  * the receive queue, but we do know that the total number of bytes queued does
260  * not exceed the receive buffer size, so we pull packets until none are left
261  * or we've read that many bytes. */
262 int
263 drain_rcvbuf(int fd)
264 {
265  int rcvbuf;
266 
267  rcvbuf = get_socket_rcvbuf(fd);
268  if (rcvbuf < 0) {
269  return -rcvbuf;
270  }
271 
272  while (rcvbuf > 0) {
273  /* In Linux, specifying MSG_TRUNC in the flags argument causes the
274  * datagram length to be returned, even if that is longer than the
275  * buffer provided. Thus, we can use a 1-byte buffer to discard the
276  * incoming datagram and still be able to account how many bytes were
277  * removed from the receive buffer.
278  *
279  * On other Unix-like OSes, MSG_TRUNC has no effect in the flags
280  * argument. */
281  char buffer[LINUX_DATAPATH ? 1 : 2048];
282  ssize_t n_bytes = recv(fd, buffer, sizeof buffer,
283  MSG_TRUNC | MSG_DONTWAIT);
284  if (n_bytes <= 0 || n_bytes >= rcvbuf) {
285  break;
286  }
287  rcvbuf -= n_bytes;
288  }
289  return 0;
290 }

563  * 'target', which should be a string in the format "<host>[:<port>]". <host>
564  * is required. If 'default_port' is nonzero then <port> is optional and
565  * defaults to 'default_port'.
566  *
567  * 'style' should be SOCK_STREAM (for TCP) or SOCK_DGRAM (for UDP).
568  *
569  * On success, returns 0 (indicating connection complete) or EAGAIN (indicating
570  * connection in progress), in which case the new file descriptor is stored
571  * into '*fdp'. On failure, returns a positive errno value other than EAGAIN
572  * and stores -1 into '*fdp'.
573  *
574  * If 'sinp' is non-null, then on success the target address is stored into
575  * '*sinp'.
576  *

578  * should be in the range [0, 63] and will automatically be shifted to the
579  * appropriately place in the IP tos field. */

581 inet_open_active(int style, const char *target, uint16_t default_port,

583 {

585  int fd = -1;
586  int error;
587 
588  /* Parse. */
589  if (!inet_parse_active(target, default_port, &sin)) {
590  error = EAFNOSUPPORT;
591  goto exit;
592  }
593 
594  /* Create non-blocking socket. */
595  fd = socket(AF_INET, style, 0);
596  if (fd < 0) {
597  /* VLOG_ERR("%s: socket: %s", target, ovs_strerror(errno));*/
598  error = errno;
599  goto exit;
600  }
601  error = set_nonblocking(fd);
602  if (error) {
603  goto exit;
604  }
605 
606  /* The dscp bits must be configured before connect() to ensure that the TOS
607  * field is set during the connection establishment. If set after
608  * connect(), the handshake SYN frames will be sent with a TOS of 0. */
609  error = set_dscp(fd, dscp);
610  if (error) {
611  /* VLOG_ERR("%s: socket: %s", target, ovs_strerror(error));*/
612  goto exit;
613  }
614 
615  /* Connect. */
616  error = connect(fd, (struct sockaddr *) &sin, sizeof sin) == 0 ? 0 : errno;
617  if (error == EINPROGRESS) {
618  error = EAGAIN;
619  }
620 
621 exit:
622  if (!error || error == EAGAIN) {
623  if (sinp) {
624  *sinp = sin;
625  }
626  } else if (fd >= 0) {
627  close(fd);
628  fd = -1;
629  }
630  *fdp = fd;
631  return error;
632 }
