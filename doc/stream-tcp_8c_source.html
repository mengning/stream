43 static int
44 new_tcp_stream(const char *name, int fd, int connect_status,
45  const struct sockaddr_in *remote, struct stream **streamp)
46 {
47  struct sockaddr_in local;
48  socklen_t local_len = sizeof local;
49  int on = 1;
50  int retval;
51 
52  /* Get the local IP and port information */
53  retval = getsockname(fd, (struct sockaddr *)&local, &local_len);
54  if (retval) {
55  memset(&local, 0, sizeof local);
56  }
57 
58  retval = setsockopt(fd, IPPROTO_TCP, TCP_NODELAY, &on, sizeof on);
59  if (retval) {
60  /* VLOG_ERR("%s: setsockopt(TCP_NODELAY): %s", name, ovs_strerror(errno));*/
61  close(fd);
62  return errno;
63  }
64 
65  retval = new_fd_stream(name, fd, connect_status, streamp);
66  if (!retval) {
67  struct stream *stream = *streamp;
68  stream_set_remote_ip(stream, remote->sin_addr.s_addr);
69  stream_set_remote_port(stream, remote->sin_port);
70  stream_set_local_ip(stream, local.sin_addr.s_addr);
71  stream_set_local_port(stream, local.sin_port);
72  }
73  return retval;
74 }

76 static int
77 tcp_open(const char *name, char *suffix, struct stream **streamp, uint8_t dscp)
78 {
79  struct sockaddr_in sin;
80  int fd, error;
81 
82  error = inet_open_active(SOCK_STREAM, suffix, 0, &sin, &fd, dscp);
83  if (fd >= 0) {
84  return new_tcp_stream(name, fd, error, &sin, streamp);
85  } else {
86  /* VLOG_ERR("%s: connect: %s", name, ovs_strerror(error)); */
87  return error;
88  }
89 }

109 static int
110 ptcp_open(const char *name OVS_UNUSED, char *suffix, struct pstream **pstreamp,
111  uint8_t dscp)
112 {
113  struct sockaddr_in sin;
114  char bound_name[128];
115  int error;
116  int fd;
117 
118  fd = inet_open_passive(SOCK_STREAM, suffix, -1, &sin, dscp);
119  if (fd < 0) {
120  return -fd;
121  }
122 
123  sprintf(bound_name, "ptcp:%"PRIu16":"IP_FMT,
124  ntohs(sin.sin_port), IP_ARGS(sin.sin_addr.s_addr));
125  error = new_fd_pstream(bound_name, fd, ptcp_accept, set_dscp, NULL,
126  pstreamp);
127  if (!error) {
128  pstream_set_bound_port(*pstreamp, sin.sin_port);
129  }
130  return error;
131 }

133 static int
134 ptcp_accept(int fd, const struct sockaddr *sa, size_t sa_len,
135  struct stream **streamp)
136 {
137  const struct sockaddr_in *sin = ALIGNED_CAST(const struct sockaddr_in *,
138  sa);
139  char name[128];
140 
141  if (sa_len == sizeof(struct sockaddr_in) && sin->sin_family == AF_INET) {
142  sprintf(name, "tcp:"IP_FMT, IP_ARGS(sin->sin_addr.s_addr));
143  sprintf(strchr(name, '\0'), ":%"PRIu16, ntohs(sin->sin_port));
144  } else {
145  strcpy(name, "tcp");
146  }
147  return new_tcp_stream(name, fd, 0, sin, streamp);
148 }
