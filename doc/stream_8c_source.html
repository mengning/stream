286 scs_connecting(struct stream *stream)
287 {

298 /* Tries to complete the connection on 'stream'. If 'stream''s connection is
299  * complete, returns 0 if the connection was successful or a positive errno
300  * value if it failed. If the connection is still in progress, returns
301  * EAGAIN. */
302 int
303 stream_connect(struct stream *stream)
304 {
305  enum stream_state last_state;
306 
307  do {
308  last_state = stream->state;
309  switch (stream->state) {
310  case SCS_CONNECTING:
311  scs_connecting(stream);
312  break;
313 
314  case SCS_CONNECTED:
315  return 0;
316 
317  case SCS_DISCONNECTED:
318  return stream->error;
319 
320  default:
321  NOT_REACHED();
322  }
323  } while ((unsigned)stream->state != last_state);
324 
325  return EAGAIN;
326 }

605  * The initial connection status, supplied as 'connect_status', is interpreted
606  * as follows:
607  *
