681 /* Copies 'n_bits' bits starting from bit 'src_ofs' in 'src' to the 'n_bits'
682  * starting from bit 'dst_ofs' in 'dst'. 'src' is 'src_len' bytes long and
683  * 'dst' is 'dst_len' bytes long.
684  *
685  * If you consider all of 'src' to be a single unsigned integer in network byte
686  * order, then bit N is the bit with value 2**N. That is, bit 0 is the bit
687  * with value 1 in src[src_len - 1], bit 1 is the bit with value 2, bit 2 is
688  * the bit with value 4, ..., bit 8 is the bit with value 1 in src[src_len -
689  * 2], and so on. Similarly for 'dst'.
690  *
691  * Required invariants:
692  * src_ofs + n_bits <= src_len * 8
693  * dst_ofs + n_bits <= dst_len * 8
694  * 'src' and 'dst' must not overlap.
695  */
696 void
697 bitwise_copy(const void *src_, unsigned int src_len, unsigned int src_ofs,
698  void *dst_, unsigned int dst_len, unsigned int dst_ofs,
699  unsigned int n_bits)
700 {
701  const uint8_t *src = src_;
702  uint8_t *dst = dst_;
703 
704  src += src_len - (src_ofs / 8 + 1);
705  src_ofs %= 8;
706 
707  dst += dst_len - (dst_ofs / 8 + 1);
708  dst_ofs %= 8;
709 
710  if (src_ofs == 0 && dst_ofs == 0) {
711  unsigned int n_bytes = n_bits / 8;
712  if (n_bytes) {
713  dst -= n_bytes - 1;
714  src -= n_bytes - 1;
715  memcpy(dst, src, n_bytes);
716 
717  n_bits %= 8;
718  src--;
719  dst--;
720  }
721  if (n_bits) {
722  uint8_t mask = (1 << n_bits) - 1;
723  *dst = (*dst & ~mask) | (*src & mask);
724  }
725  } else {
726  while (n_bits > 0) {
727  unsigned int max_copy = 8 - MAX(src_ofs, dst_ofs);
728  unsigned int chunk = MIN(n_bits, max_copy);
729  uint8_t mask = ((1 << chunk) - 1) << dst_ofs;
730 
731  *dst &= ~mask;
732  *dst |= ((*src >> src_ofs) << dst_ofs) & mask;
733 
734  src_ofs += chunk;
735  if (src_ofs == 8) {
736  src--;
737  src_ofs = 0;
738  }
739  dst_ofs += chunk;
740  if (dst_ofs == 8) {
741  dst--;
742  dst_ofs = 0;
743  }
744  n_bits -= chunk;
745  }
746  }
747 }
748 
749 /* Zeros the 'n_bits' bits starting from bit 'dst_ofs' in 'dst'. 'dst' is
750  * 'dst_len' bytes long.
751  *
752  * If you consider all of 'dst' to be a single unsigned integer in network byte
753  * order, then bit N is the bit with value 2**N. That is, bit 0 is the bit
754  * with value 1 in dst[dst_len - 1], bit 1 is the bit with value 2, bit 2 is
